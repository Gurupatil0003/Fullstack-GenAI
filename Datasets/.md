```python
import seaborn as sns
import matplotlib.pyplot as plt

# Simple dummy list data
x = ['A', 'B', 'C', 'D', 'E']
y = [10, 20, 15, 25, 18]

# Repeat x and y for plots needing more data
x2 = x * 2
y2 = y + y

sns.set_theme(style="whitegrid")  # Optional: clean background

# ðŸ”¹ Bar Plot
sns.barplot(x=x, y=y)
plt.title("Bar Plot")
plt.show()

# ðŸ”¹ Line Plot
sns.lineplot(x=x, y=y)
plt.title("Line Plot")
plt.show()

# ðŸ”¹ Scatter Plot
sns.scatterplot(x=x, y=y)
plt.title("Scatter Plot")
plt.show()

# ðŸ”¹ Box Plot
sns.boxplot(x=x2, y=y2)
plt.title("Box Plot")
plt.show()

# ðŸ”¹ Strip Plot
sns.stripplot(x=x2, y=y2)
plt.title("Strip Plot")
plt.show()

# ðŸ”¹ Swarm Plot
sns.swarmplot(x=x2, y=y2)
plt.title("Swarm Plot")
plt.show()

# ðŸ”¹ Histogram
sns.histplot(y, bins=5)
plt.title("Histogram")
plt.show()

# ðŸ”¹ KDE Plot
sns.kdeplot(y, fill=True)
plt.title("KDE Plot")
plt.show()



```
